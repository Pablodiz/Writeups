import sys

def scramble(L):
  # [['0x48'], ['0x4f'], ['0x4c'], ['0x41'], ['0x41'], ['0x41'], ['0x41']]
  A = L
  i = 2
  while (i < len(A)):
    A[i-2] += A.pop(i-1)
    A[i-1].append(A[:i-2])
    i += 1
  
  # Hace una lista que es 
  # [
  #   ['0x48', '0x4f'],                           <- L0, L1 = A
  #   ['0x4c', [], '0x41'],                       <- L2, [], L3  <- B
  #   ['0x41', [['0x48', '0x4f']], '0x41'],       <- L4, [A] L5  <- C
  #   ['0x41', [['0x48', '0x4f'], ['0x4c', [], '0x41']]]]     <- L6, [A, B]
  # Por cada dos caracteres, se hace una lista en la que lo importante es que se pone 1 en la primera posición y el siguiente en la última
  return L

def de_scramble(scrambled):
  extracted = []

  for item in scrambled:
    if len(item) == 0:
        continue
    # primer y último elemento de la sublista
    first = item[0]
    last = item[-1]
    if isinstance(first, str) and first.startswith("0x"):
        extracted.append(first)
    if isinstance(last, str) and last.startswith("0x") and last != first:
        extracted.append(last)
  return extracted
